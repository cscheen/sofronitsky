\section{Disques compacts non répertoriés}
\label{sec:Nonrepertorie}

Les disques compacts suivants ne sont pas encore répertoriés~:
\begin{itemize}
 \item
 \emph{Arioso} (ARI~700, boîtier de deux disques compacts).
 Ces disques contiennent en particulier les vingt-quatre Préludes \Opus{28}
 de \Chopin{}, six Sonates de \Scriabine{} et des œuvres de \Liadov{}~;
 \item
 \emph{Melodija/Bukok} (DE~0176).
 \Scriabine{} (Sonates \Opus{23} et \Opus{30})~;
 \item
 \emph{Melodija/Bukok} (DE~0177).
 \Scriabine{} (Sonates \Opus{53}, \Opus{66}, \Opus{68} et \Opus{70})~;
 \item
 \emph{Melodija/Bukok} (DE~0178).
 \Scriabine{} (Études \Opus{42} \Number{4} et \Opus{42} \Number{6})~;
 \item
 \emph{Melodija/Bukok} (DE~0180).
 \Scriabine{} (Mazurkas \Opus{3} \Number{6} et \Opus{3} \Number{9},
 Polonaise \Opus{21}, Poèmes \Opus{32} \Number{1}, \Opus{32} \Number{2},
 \Opus{34}, \Opus{36} et \Opus{41}, Morceaux \Opus{52} \Number{1}, \Opus{52}
 \Number{2} et \Opus{52} \Number{3}, Fantaisie \Opus{28})~;
 \item
 \emph{Selene} (\hbox{CD-s}~9809.45).
 \Chopin{} (Valse \Opus{64} \Number{1}, Valse \Opus{70} \Number{1}, Préludes
 \Opus{28} \Number{13}, \Opus{28} \Number{14}, \Opus{28} \Number{20},
 \Opus{28} \Number{22}, Nouvelle Étude en \kA \Flat majeur, Étude \Opus{10}
 \Number{4})~;
 \item
 \emph{Venecija} (CDVE~05218, boîtier de deux disques compacts).
 \Scriabine{} (Préludes \Opus{11} \Number{1}, \Opus{11} \Number{13},
 \Opus{11} \Number{14}, \Opus{22} \Number{1} et \Opus{49} \Number{2},
 Morceau \Opus{57} \Number{1}, Étude \Opus{8} \Number{12}, Poème \Opus{32}
 \Number{1}, Préludes \Opus{11} \Number{2}, \Opus{11} \Number{4}, \Opus{11}
 \Number{5}, \Opus{11} \Number{9}, \Opus{11} \Number{10}, \Opus{11}
 \Number{15}, \Opus{11} \Number{16}, \Opus{11} \Number{17}, \Opus{11}
 \Number{19}, \Opus{11} \Number{20}, \Opus{11} \Number{21}, \Opus{11}
 \Number{22}, \Opus{11} \Number{23} et \Opus{11} \Number{24}, Études
 \Opus{8} \Number{2}, \Opus{8} \Number{4}, \Opus{8} \Number{5}, \Opus{8}
 \Number{8}, \Opus{8} \Number{9}, \Opus{8} \Number{11} et \Opus{8}
 \Number{12}, Sonates \Opus{6}, \Opus{23}, \Opus{30}, \Opus{53}, \Opus{66},
 \Opus{68} et \Opus{70}).
\end{itemize}
